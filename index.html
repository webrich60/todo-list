/** * TODO管理アプリ用 バックエンドプログラム (双方向同期・自動カレンダー＆メール対応・JSONP対応・超堅牢版) */ function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const data = sheet.getDataRange().getValues(); const callback = e.parameter.callback; let resultJSON = "[]"; if (data.length > 1) { const headers = data[0]; const idCol = headers.indexOf('ID'); const typeCol = headers.indexOf('種類'); const titleCol = headers.indexOf('タスク名'); const descCol = headers.indexOf('詳細'); const priorityCol = headers.indexOf('優先度'); const dateCol = headers.indexOf('開始日時'); const endDateCol = headers.indexOf('終了日時'); const colorCol = headers.indexOf('カラー'); const statusCol = headers.indexOf('ステータス'); const mailNotifyCol = headers.indexOf('メール通知'); const todos = []; for (let i = 1; i < data.length; i++) { const row = data[i]; if (!row[idCol]) continue; todos.push({ id: Number(row[idCol]), type: row[typeCol] === 'メモ' ? 'memo' : 'todo', title: String(row[titleCol]), description: String(row[descCol] || ''), priority: String(row[priorityCol] || '中'), dueDate: row[dateCol] && row[dateCol] !== '期限なし' ? formatDateToISO(row[dateCol]) : null, endDate: row[endDateCol] ? formatDateToISO(row[endDateCol]) : null, color: String(row[colorCol] || 'blue'), completed: row[statusCol] === '完了', mailNotify: row[mailNotifyCol] === '希望する' }); } resultJSON = JSON.stringify(todos); } if (callback) { return ContentService.createTextOutput(callback + '(' + resultJSON + ');') .setMimeType(ContentService.MimeType.JAVASCRIPT); } else { return ContentService.createTextOutput(resultJSON) .setMimeType(ContentService.MimeType.JSON); } } function formatDateToISO(dateVal) { if (dateVal instanceof Date) { return Utilities.formatDate(dateVal, "Asia/Tokyo", "yyyy-MM-dd'T'HH:mm"); } return dateVal; } function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); let data; try { data = JSON.parse(e.postData.contents); } catch (error) { return ContentService.createTextOutput("Error: データの解析に失敗しました。"); } if (!data || data.length === 0) { return ContentService.createTextOutput("No data to update."); } if (sheet.getLastRow() === 0) { sheet.appendRow(['ID', '種類', 'タスク名', '詳細', '優先度', '開始日時', '終了日時', 'カラー', 'ステータス', 'メール通知', 'カレンダー連携']); } const calendar = CalendarApp.getDefaultCalendar(); const lastRow = sheet.getLastRow(); let existingValues = []; if (lastRow > 1) { existingValues = sheet.getRange(2, 1, lastRow - 1, 1).getValues(); } const idRowMap = {}; for (let i = 0; i < existingValues.length; i++) { const idVal = existingValues[i][0]; if (idVal) { idRowMap[idVal.toString()] = i + 2; } } data.forEach(function(todo) { let calendarStatus = "対象外"; // 完了ステータスの解釈を文字列・真偽値の両方に対応（超堅牢化） const isCompleted = todo.completed === true || todo.completed === '完了'; if (todo.dueDate && todo.dueDate !== '期限なし') { try { const startDate = new Date(todo.dueDate); let endDate; if (todo.endDate && todo.endDate !== '') { endDate = new Date(todo.endDate); if (endDate <= startDate) { endDate = new Date(startDate.getTime() + 60 * 60 * 1000); } } else { endDate = new Date(startDate.getTime() + 60 * 60 * 1000); } const isMemo = todo.type === 'memo'; const typeMark = isMemo ? '📝メモ' : 'TODO'; const priorityMark = todo.priority ? `【${todo.priority}】` : '【中】'; const activeTitle = `${typeMark}: ${priorityMark}${todo.title}`; const doneTitle = `✅完了: ${todo.title}`; const description = todo.description || ""; let eventColor = null; if (todo.color === 'green') eventColor = CalendarApp.EventColor.PALE_GREEN; else if (todo.color === 'yellow') eventColor = CalendarApp.EventColor.YELLOW; else if (todo.color === 'purple') eventColor = CalendarApp.EventColor.MAUVE; else if (todo.color === 'orange') eventColor = CalendarApp.EventColor.PALE_RED; else eventColor = CalendarApp.EventColor.PALE_BLUE; const events = calendar.getEvents(startDate, endDate); let targetEvent = null; for (let i = 0; i < events.length; i++) { const t = events[i].getTitle(); if (t.includes(todo.title)) { targetEvent = events[i]; break; } } if (isCompleted) { if (targetEvent) { targetEvent.setTitle(doneTitle); targetEvent.setDescription(description); calendarStatus = "完了更新済"; } } else { if (!targetEvent) { const newEvent = calendar.createEvent(activeTitle, startDate, endDate); newEvent.setDescription(description); if (eventColor) newEvent.setColor(eventColor); calendarStatus = "新規登録済"; } else { targetEvent.setTitle(activeTitle); targetEvent.setTime(startDate, endDate); targetEvent.setDescription(description); if (eventColor) targetEvent.setColor(eventColor); calendarStatus = "同期済"; } } } catch (err) { calendarStatus = "エラー: " + err.message; } } const rowData = [ todo.id, todo.type === 'memo' ? 'メモ' : 'TODO', todo.title, todo.description || '', todo.priority || '中', todo.dueDate, todo.endDate || '', todo.color || 'blue', isCompleted ? '完了' : '未完了', todo.mailNotify ? '希望する' : '希望しない', calendarStatus ]; const bgColor = isCompleted ? '#fce4ec' : '#ffffff'; const bgRow = new Array(rowData.length).fill(bgColor); try { const todoIdStr = todo.id.toString(); if (idRowMap[todoIdStr]) { const rowNum = idRowMap[todoIdStr]; const range = sheet.getRange(rowNum, 1, 1, rowData.length); range.setValues([rowData]); range.setBackgrounds([bgRow]); } else { sheet.appendRow(rowData); const newRowNum = sheet.getLastRow(); const range = sheet.getRange(newRowNum, 1, 1, rowData.length); range.setBackgrounds([bgRow]); idRowMap[todoIdStr] = newRowNum; } } catch (err) { sheet.appendRow(['エラー', '書き込み失敗', err.message, '', '', '', '', '', '', '', '']); } }); try { organizeDriveFiles(sheet); } catch (err) { console.error("ドライブ整理エラー: " + err); } return ContentService.createTextOutput("Success"); } function organizeDriveFiles(sheet) { const folderName = "TODO管理アプリ_データ"; let folder; const folders = DriveApp.getFoldersByName(folderName); if (folders.hasNext()) { folder = folders.next(); } else { folder = DriveApp.createFolder(folderName); } const ss = SpreadsheetApp.getActiveSpreadsheet(); const file = DriveApp.getFileById(ss.getId()); let inFolder = false; const parents = file.getParents(); while (parents.hasNext()) { if (parents.next().getId() === folder.getId()) { inFolder = true; break; } } if (!inFolder) { file.moveTo(folder); } const now = new Date(); const formattedDate = Utilities.formatDate(now, "Asia/Tokyo", "yyyy-MM-dd_HH-mm-ss"); const csvFileName = "TODOリスト_バックアップ_" + formattedDate + ".csv"; let csvContent = ""; const data = sheet.getDataRange().getValues(); data.forEach(function(row) { const rowString = row.map(function(cell) { const cellStr = String(cell).replace(/"/g, '""'); return `"${cellStr}"`; }).join(","); csvContent += rowString + "\r\n"; }); folder.createFile(csvFileName, csvContent, MimeType.CSV); } function checkAndSendReminders() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return; const headers = data[0]; const statusCol = headers.indexOf('ステータス'); const mailNotifyCol = headers.indexOf('メール通知'); const dateCol = headers.indexOf('開始日時'); const titleCol = headers.indexOf('タスク名'); const descCol = headers.indexOf('詳細'); if (statusCol === -1 || mailNotifyCol === -1 || dateCol === -1) { console.log("必要な列が見つかりません。"); return; } const userEmail = Session.getActiveUser().getEmail(); const now = new Date(); for (let i = 1; i < data.length; i++) { const row = data[i]; const status = row[statusCol]; const mailNotify = row[mailNotifyCol]; const dueDateStr = row[dateCol]; if (status !== '未完了' || mailNotify !== '希望する' || !dueDateStr || dueDateStr === '期限なし') { continue; } const dueDate = new Date(dueDateStr); const diffHours = (dueDate.getTime() - now.getTime()) / (1000 * 60 * 60); if (diffHours > 47 && diffHours <= 48) { sendAlertEmail(userEmail, row[titleCol], row[descCol], dueDateStr, "2日前"); } else if (diffHours > 2 && diffHours <= 3) { sendAlertEmail(userEmail, row[titleCol], row[descCol], dueDateStr, "3時間前"); } } } function sendAlertEmail(to, title, description, dueDateStr, timing) { const formattedDate = Utilities.formatDate(new Date(dueDateStr), "Asia/Tokyo", "yyyy/MM/dd HH:mm"); const subject = `【リマインド】期日の${timing}です: ${title}`; const body = ` 相棒、予定の${timing}になったのでお知らせします！ ■ タスク名: ${title} ■ 期日: ${formattedDate} ■ 詳細: ${description || 'なし'} TODOアプリを開いて、状況を確認してくださいね。 ゆずより `; MailApp.sendEmail(to, subject, body); }
